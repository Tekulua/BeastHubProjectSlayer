-- Função para verificar se a chave inserida e o nome do jogador são válidos
local function IsValidUser(playerName, inputKey)
    for _, user in ipairs(validUsers) do
        if user.name == playerName and user.key == inputKey then
            return true
        end
    end
    return false
end

-- Função para tocar som ao clicar no botão
local function PlayClickSound(soundId)
    local clickSound = Instance.new("Sound", SoundService)
    clickSound.SoundId = soundId
    clickSound.Volume = 1
    clickSound:Play()
    clickSound.Ended:Connect(function()
        clickSound:Destroy()
    end)
end
